-- =====================================================
-- 1. 通用函数
-- =====================================================

create or replace function public.set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

create or replace function public.handle_new_user()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
begin
  insert into public.user_profiles (
    id,
    email,
    display_name,
    avatar_url,
    locale,
    timezone,
    theme,
    status
  )
  values (
    new.id,
    new.email,
    coalesce(new.raw_user_meta_data ->> 'display_name', split_part(coalesce(new.email, ''), '@', 1)),
    new.raw_user_meta_data ->> 'avatar_url',
    'zh-CN',
    'Asia/Shanghai',
    'dark',
    'active'
  )
  on conflict (id) do nothing;

  insert into public.user_watchlists (
    user_id,
    watchlist_name,
    is_default
  )
  values (
    new.id,
    '默认自选',
    true
  )
  on conflict (user_id, watchlist_name) do nothing;

  return new;
end;
$$;

create table if not exists public.user_profiles (
  id                uuid primary key references auth.users(id) on delete cascade,
  email             text,
  display_name      text,
  avatar_url        text,
  locale            text not null default 'zh-CN',
  timezone          text not null default 'Asia/Shanghai',
  theme             text not null default 'dark',
  status            text not null default 'active',
  created_at        timestamptz not null default now(),
  updated_at        timestamptz not null default now()
);

create unique index if not exists idx_user_profiles_email_unique
  on public.user_profiles(email)
  where email is not null;

drop trigger if exists trg_user_profiles_updated_at on public.user_profiles;
create trigger trg_user_profiles_updated_at
before update on public.user_profiles
for each row
execute function public.set_updated_at();

drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
after insert on auth.users
for each row
execute function public.handle_new_user();

create table if not exists public.markets (
  id                bigserial primary key,
  market_code       text not null unique,   -- stock / crypto / forex / index / commodity
  market_name       text not null,
  sort_order        integer not null default 0,
  is_active         boolean not null default true,
  created_at        timestamptz not null default now()
);

create table if not exists public.assets (
  id                bigserial primary key,
  market_id         bigint not null references public.markets(id) on delete restrict,
  asset_code        text not null,          -- AAPL / BTCUSDT / SPX
  asset_symbol      text not null,
  asset_name        text not null,
  asset_type        text not null,          -- stock / coin / forex / index / etf / commodity

  base_currency     text,
  quote_currency    text,
  sector_name       text,
  industry_name     text,
  logo_url          text,
  description       text,
  status            text not null default 'active',
  metadata          jsonb not null default '{}'::jsonb,
  created_at        timestamptz not null default now(),
  updated_at        timestamptz not null default now(),

);

before update on public.assets
for each row

create or replace view public.v_asset_overview as
select
  a.id,
  a.market_id,
  m.market_code,
  m.market_name,
  a.asset_code,
  a.asset_symbol,
  a.asset_name,
  a.asset_type,
  a.exchange_name,
  a.base_currency,
  a.quote_currency,
  a.sector_name,
  a.industry_name,
  a.logo_url,
  a.status,
  q.last_price,
  q.prev_close_price,
  q.open_price,
  q.high_price,
  q.low_price,
  q.price_change,
  q.price_change_pct,
  q.volume,
  q.turnover,
  q.market_cap,
  q.amplitude_pct,
  q.heat_score,
  q.sentiment_score,
  q.ai_signal_label,
  q.updated_at as quote_updated_at
from public.assets a
join public.markets m on m.id = a.market_id
left join public.asset_quote_latest q on q.asset_id = a.id;

create or replace view public.v_watchlist_items as
select
  wi.id as watchlist_item_id,
  w.id as watchlist_id,
  w.user_id,
  w.watchlist_name,
  w.is_default,
  wi.sort_order,
  wi.created_at as added_at,
  a.id as asset_id,
  a.asset_code,
  a.asset_symbol,
  a.asset_name,
  a.asset_type,
  q.last_price,
  q.price_change,
  q.price_change_pct,
  q.heat_score,
  q.ai_signal_label,
  q.updated_at as quote_updated_at
from public.user_watchlist_items wi
join public.user_watchlists w on w.id = wi.watchlist_id
join public.assets a on a.id = wi.asset_id
left join public.asset_quote_latest q on q.asset_id = a.id;

create policy "ai_conversations own delete"
on public.ai_conversations

to authenticated
using (auth.uid() = user_id);

drop policy if exists "ai_messages own select" on public.ai_messages;
create policy "ai_messages own select"
on public.ai_messages
for select
to authenticated
using (
  exists (
    select 1
    from public.ai_conversations c
    where c.id = ai_messages.conversation_id
      and c.user_id = auth.uid()
  )
);

drop policy if exists "ai_messages own insert" on public.ai_messages;
create policy "ai_messages own insert"
on public.ai_messages
for insert
to authenticated
with check (
  exists (
    select 1
    from public.ai_conversations c
    where c.id = ai_messages.conversation_id
      and c.user_id = auth.uid()
  )
);
